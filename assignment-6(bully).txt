import java.util.Scanner;

public class Bully {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        int n;

        System.out.print("Enter number of processes: ");
        n = sc.nextInt();

        int processes[] = new int[n];

        for(int i = 0; i < n; i++) {
            processes[i] = i + 1;
        }

        System.out.print("Enter failed coordinator process: ");
        int failed = sc.nextInt();

        System.out.println("Process " + failed + " has failed");

        int newCoordinator = 0;

        for(int i = 0; i < n; i++) {
            if(processes[i] > failed) {
                newCoordinator = processes[i];
            }
        }

        System.out.println("New Coordinator is Process " + newCoordinator);

        sc.close();
    }
}
