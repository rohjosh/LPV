import java.util.Scanner;

public class TokenRing {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        int n;

        System.out.print("Enter number of processes: ");
        n = sc.nextInt();

        int token = 0;

        for(int i = 0; i < n; i++) {

            System.out.println("\nProcess P" + i + " has token");

            System.out.print("Does P" + i + " want to enter Critical Section? (1/0): ");
            int choice = sc.nextInt();

            if(choice == 1) {
                System.out.println("P" + i + " is entering Critical Section");
                System.out.println("P" + i + " is leaving Critical Section");
            }
            else {
                System.out.println("P" + i + " does not want Critical Section");
            }

            token = (token + 1) % n;
        }

        sc.close();
    }
}
